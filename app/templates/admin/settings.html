Modify Scheduler Settings

To change scheduler settings, edit instance/codex.conf and restart Codex:

[scheduler]
sync_freshservice_enabled = true
sync_datto_enabled = true
sync_tickets_enabled = false
sync_freshservice_schedule = daily  # daily, hourly, or disabled
sync_datto_schedule = daily
sync_tickets_schedule = hourly
sync_run_on_startup = false
